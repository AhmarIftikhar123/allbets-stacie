// --- Shortcode: [product_grid] ---
// Display Products (from "ms_product" post type) in a responsive card grid.

// Usage examples:

// [product_grid]
// -> default 6 products, 3 columns, pagination ON

// [product_grid count="9" columns="3"]
// -> show 9 products, 3 columns grid

// [product_grid count="12" columns="4" paged="false"]
// -> 12 products, 4 columns, NO pagination

// [product_grid paged="true"]
// -> enable pagination (default enabled)

// ---- Attributes Reference ----
// count="6"          number of products to show
// columns="3"        2, 3, or 4 column grid using theme's ms-cols-* classes
// paged="true/false" enable or disable pagination

// ---- Notes ----
// • Card layout is loaded from: template-parts/products/content-product-card.php
// • Grid styling uses existing theme classes: .ms-cards + .ms-cols-*
// • Products support: title, featured image, description, external link (via meta box)
